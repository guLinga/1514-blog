方法一：暴力破解
```js
var twoSum = function(nums, target) {
    let map = new Map();
    for(let i=0,len=nums.length;i<len;i++){
        let ans = map.get(target-nums[i])
        if(ans!=undefined){
            return [ans,i];
        }else{
            map.set(nums[i],i);
        }
    }
};
```
方法二：Map
```js
var twoSum = function(nums, target) {
    for(let i=0;i<nums.length;i++){
        for(let j=i+1;j<nums.length;j++){
            if(nums[i]+nums[j]===target)
            return [i,j]
        }
    }
};
```
