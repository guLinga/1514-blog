```js
var countSubstrings = function(s) {
  let num = 0;
  for(let i=0;i<s.length;i++){
    num += reverse(s,i,i);
    num += reverse(s,i,i+1);
  }
  return num;
};
function reverse(s,left,right){
  let num = 0;
  while(left>=0&&right<=s.length-1&&s[left]===s[right]){
    num++;
    left--;
    right++;
  }
  return num;
}
```
