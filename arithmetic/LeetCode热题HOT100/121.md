[题目](https://leetcode.cn/problems/best-time-to-buy-and-sell-stock/)

```js
var maxProfit = function(prices) {
  let ans = 0;
  let idx = 0;
  let left = idx;
  while(idx<prices.length){
    let temp = prices[idx] - prices[left];
    if(temp>0){
      ans = Math.max(ans, temp);
    }else{
      left = idx;
    }
    idx++;
  }
  return ans;
};
```
