我也不知道叫什么方法，我自己想出来的，遍历一遍nums数组的每个元素，每次遍历nums元素的时候就重新遍历一下arr数组，深拷贝一下arr数组的元素，想arr数组的元素中添加nums[i]
```js
var subsets = function(nums) {
  let arr = [[]];
  let len = nums.length;
  for(let i=0;i<nums.length;i++){
    let len = arr.length;
    for(let j=0;j<len;j++){
      let temp = JSON.parse(JSON.stringify(arr[j]));
      temp.push(nums[i]);
      arr.push(temp);
    }
  }
  return arr;
};
```
