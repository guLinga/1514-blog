[题目](https://leetcode.cn/problems/daily-temperatures/)

方法一：暴力破解
```js
var dailyTemperatures = function(temperatures) {
  const len = temperatures.length;
  let stack = [];
  for(let i=0;i<len;i++){
    let idx = i + 1;
    while(idx<len&&temperatures[idx]<=temperatures[i]){
      idx++;
    }
    if(idx>=len)stack.push(0);
    else
    stack.push(idx - i);
  }
  return stack;
};
```
