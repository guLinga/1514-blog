[题目](https://leetcode.cn/problems/merge-k-sorted-lists/)


方法一：暴力解法，使用循环遍历每一个链表，然后两个两个的合并。
```js
var mergeKLists = function(lists) {
  if(lists.length===0)return null;
  let ans = lists[0];
  for(let i=1;i<lists.length;i++){
    ans = List(ans,lists[i]);
  }
  function List(l1,l2){
    if(l1===null&&l2===null)return null;
    if(l1===null)return l2;
    if(l2===null)return l1;
    if(l1.val<l2.val){
      l1.next = List(l1.next,l2);
      return l1;
    }else{
      l2.next = List(l1,l2.next);
      return l2;
    }
  }
  return ans;
};
```
方法二：分治算法
```js
var mergeKLists = function(lists) {
  if(lists.length===0)return null;
  let len = lists.length;
  let init = 1;
  while(init<len){
    for(let i=0;i<len;i+=init*2){
      if(i+init<len){
        lists[i] = List(lists[i],lists[i+init]);
      }
    }
    init*=2;
  }
  function List(l1,l2){
    if(l1===null&&l2===null)return null;
    if(l1===null)return l2;
    if(l2===null)return l1;
    if(l1.val<l2.val){
      l1.next = List(l1.next,l2);
      return l1;
    }else{
      l2.next = List(l1,l2.next);
      return l2;
    }
  }
  return lists[0];
};
```
