使用shift函数好像时间就上去了
```js
var singleNumber = function(nums) {
  nums.sort((x,y)=>x-y);
  while(nums.length){
    let shift1 = nums.shift();
    let shift2 = nums.shift();
    if(shift1!==shift2)return shift1;
  }
};
```

```js
var singleNumber = function(nums) {
  nums.sort((x,y)=>x-y);
  for(let i=1;i<nums.length;i+=2){
    if(nums[i]!==nums[i-1]){
      return nums[i-1];
    }
  }
  return nums[nums.length-1];
};
```
