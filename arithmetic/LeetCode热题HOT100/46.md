var permute = function(nums) {
  if(nums.length===1)return [nums];
  let arr = [];
  function dfs(aim,nums){
    if(nums.length===0){
      arr.push(aim);
      return;
    }
    for(let i=0;i<nums.length;i++){
      let temp = [...nums];
      let num = temp.splice(i,1);
      dfs([...aim,num],temp);
    }
  }
  dfs([],nums);
  return arr;
};
