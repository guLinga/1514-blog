[题目](https://leetcode.cn/problems/target-sum/)


方法一：递归
```js
var findTargetSumWays = function(nums, target) {
  let num = 0;
  function get(nums,ans,idx){
    if(ans===target&&idx===nums.length){
      num++;
      return;
    }
    if(idx>nums.length)return;
    get(nums,ans+nums[idx],idx+1);
    get(nums,ans-nums[idx],idx+1);
  }
  get(nums,0,0);
  return num;
};
```
