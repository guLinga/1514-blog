```js
var findUnsortedSubarray = function(nums) {
  let len = nums.length;
  let min = nums[len-1];
  let max = nums[0];
  let start = 0,end = -1;
  for(let i=0;i<len;i++){
    if(nums[i]<max){
      end = i;
    }else{
      max = nums[i];
    }

    if(nums[len-i-1]>min){
      start = len-i-1;
    }else{
      min = nums[len-i-1];
    }
  }
  return end-start+1;
};
```
